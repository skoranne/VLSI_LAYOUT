Using the hash_table object design_hash_table we convert d1 and o1 into design_dbs(1)
and design_dbs(2) to get a design object. As you can see each design
type also has a hash_table, and using that hash table we can conver
a string like d1:poly to a type(Layer) using the following steps:

Step 1: use the design_hash_table to convert string d1 into db_index

Step 2: index into design_dbs array to get a type(Design)

Step 3: call hash_get( design_dbs(db_index)%ht, layer_name ) to get layer index

Step 4: index into design_dbs(db_index)%layers(layer_index) to get array

pure subroutine GetLayerFromDBName( full_name, design_hash_table, design_dbs )
  character(len=*), intent(in) :: full_name
  type(hash_type), intent(in) :: design_hash_table
  type(Design),intent(in)     :: design_dbs(:)
  character(len=*), allocatable:: layerName
  integer                     :: db_index, layer_index
  logical :: ins
  integer :: pos
  pos = index( full_name, ":")
  call hash_get( design_hash_table, full_name(1:pos), db_index, ins )
  layerName = trim( full_name(1+pos,:) )
  call hash_get( design_dbs(db_index)%ht, layerName, layer_index, ins )
end subroutine GetLayerFromDBName
